function DashboardPage({ indData, setIndData, user, company }) {

  const totalDone = INDICATORS.filter(isDone).length;
  const totalPct  = Math.round((totalDone/INDICATORS.length)*100);

  // 이행 기록 추가 (내용 입력 + 선택적 파일)
  const addRecord = async (ind, content, file) => {
    let fileName=null, fileData=null, fileType=null;
    if (file) {
      fileName = file.name;
      fileData = await readFileAsBase64(file);
      fileType = file.type;
    }
    setIndData(prev => {
      const recs = prev[ind.id]?.records || [];
      if (recs.length >= ind.target) return prev; // 이미 완료
      return { ...prev, [ind.id]: { ...prev[ind.id], records: [...recs, { id:Date.now(), date:today(), content, fileName, fileData, fileType }] } };
    });
  };

  const delRecord = (indId, recId) => {
    setIndData(prev => ({ ...prev, [indId]: { ...prev[indId], records: prev[indId].records.filter(r=>r.id!==recId) } }));
  };

    <div>

        <StatCard label="이행 완료"     value={`${totalDone}/${INDICATORS.length}`} color={C.success} />
        <StatCard label="진행 중"       value={INDICATORS.filter(i=>getRecords(i.id).length>0&&!isDone(i)).length} color={C.warning} />
        <StatCard label="미이행"        value={INDICATORS.filter(i=>getRecords(i.id).length===0).length} color={C.danger} />
        <button onClick={()=>setReportModal(true)} style={{ ...BS,background:C.purple,color:"#fff",fontWeight:700,fontSize:12 }}>📄 경영책임자 보고서</button>

      {/* 지표 그룹 */}
      {groups.map(grp=>(
        <div key={grp} style={{ marginBottom:20 }}>
          <div style={{ fontSize:13,fontWeight:700,color:C.accent,marginBottom:10,padding:"4px 10px",background:C.accentDim,borderRadius:6,display:"inline-block" }}>{grp}</div>
          <div style={{ display:"flex",flexDirection:"column",gap:8 }}>
            {INDICATORS.filter(i=>i.group===grp).map(ind=>{
              const recs = getRecords(ind.id);
              const done = isDone(ind);
              const pct  = Math.min(100,Math.round((recs.length/ind.target)*100));
              return (
                <div key={ind.id} style={{ background:C.surface,border:`1px solid ${done?C.success+"55":C.border}`,borderRadius:10,padding:"12px 14px" }}>
                  <div style={{ display:"flex",alignItems:"center",gap:8,flexWrap:"wrap" }}>
                    <span style={{ fontSize:12,fontWeight:700,color:done?C.success:C.text,flex:1 }}>{ind.label}</span>
                    <span style={{ fontSize:11,color:C.textMuted,whiteSpace:"nowrap" }}>{ind.period}</span>
                    <span style={{ fontSize:13,fontWeight:800,color:done?C.success:recs.length>0?C.warning:C.danger,whiteSpace:"nowrap" }}>
                      {recs.length}/{ind.target}
                    </span>
                    {done && <span style={{ ...BDG,background:C.successDim,color:C.success }}>✓ 완료</span>}
                    {!done && recs.length===0 && <span style={{ ...BDG,background:C.dangerDim,color:C.danger }}>미이행</span>}
                    {!done && recs.length>0 && <span style={{ ...BDG,background:C.warningDim,color:C.warning }}>진행중</span>}
                  </div>

                  {/* 진행 바 */}
                  <div style={{ height:6,background:C.surface2,borderRadius:99,margin:"8px 0 4px" }}>
                    <div style={{ height:"100%",width:`${pct}%`,background:done?C.success:recs.length>0?C.warning:C.border,borderRadius:99,transition:"width .4s" }} />
                  </div>

                  {/* 이행 기록 미리보기 */}
                  {recs.length>0 && (
                    <div style={{ marginBottom:6,display:"flex",flexDirection:"column",gap:3 }}>
                      {recs.map((r,idx)=>(
                        <div key={r.id} style={{ display:"flex",alignItems:"center",gap:6,fontSize:11,color:C.textSub }}>
                          <span style={{ color:C.success }}>●</span>
                          <span style={{ color:C.textMuted,whiteSpace:"nowrap" }}>{r.date}</span>
                          <span style={{ flex:1,overflow:"hidden",textOverflow:"ellipsis",whiteSpace:"nowrap" }}>{r.content}</span>
                          {r.fileName && (
                            <button onClick={()=>setViewModal({ ind, rec:r })} style={{ ...SB,color:C.info,fontSize:10,padding:"2px 6px" }}>📎 {r.fileName.length>12?r.fileName.slice(0,12)+"…":r.fileName}</button>
                          )}
                          {user.role==="master" && <button onClick={()=>delRecord(ind.id,r.id)} style={{ ...SB,color:C.danger,fontSize:10,padding:"2px 6px" }}>✕</button>}
                        </div>
                      ))}
                    </div>
                  )}

                  {/* 액션 버튼 */}
                  <div style={{ display:"flex",gap:6,flexWrap:"wrap",marginTop:4 }}>
                    <button onClick={()=>setLawModal(ind)} style={{ ...SB,color:C.info }}>📋 법령확인</button>
                    {recs.length>0 && <button onClick={()=>setViewModal({ ind, rec:null })} style={{ ...SB,color:C.success }}>📂 기록 전체보기</button>}
                    {user.role==="master" && !done && (
                      <button onClick={()=>setRecordModal(ind)} style={{ ...SB,color:C.accent,fontWeight:700,borderColor:C.accent }}>
                        ✏️ 이행 기록 추가 ({ind.target-recs.length}회 남음)
                      </button>
                    )}
                    {user.role==="master" && done && (
                      <span style={{ fontSize:11,color:C.success }}>✅ 이행 완료 — 기록 {recs.length}건</span>
                    )}
                  </div>
                </div>
              );
            })}
          </div>
        </div>
      ))}

      {/* ── 이행기록 추가 모달 ── */}
      {recordModal && (
        <RecordAddModal
          ind={recordModal}
          currentCount={(indData[recordModal.id]?.records||[]).length}
          onAdd={(content, file) => { addRecord(recordModal, content, file); setRecordModal(null); }}
          onClose={()=>setRecordModal(null)}
        />
      )}

      {/* ── 파일 및 기록 전체보기 모달 ── */}
      {viewModal && (
        <RecordViewModal
          ind={viewModal.ind}
          singleRec={viewModal.rec}
          records={indData[viewModal.ind.id]?.records||[]}
          signatureName={indData[viewModal.ind.id]?.signatureName||""}
          signatureDate={indData[viewModal.ind.id]?.signatureDate||""}
          onSign={(name,date)=>setIndData(prev=>({...prev,[viewModal.ind.id]:{...prev[viewModal.ind.id],signatureName:name,signatureDate:date}}))}
          onClose={()=>setViewModal(null)}
          company={company}
        />
      )}

      {/* ── 법령 모달 ── */}
      {lawModal && (
        <Modal onClose={()=>setLawModal(null)} title="📋 관계법령 확인">
          <div style={{ fontSize:13,lineHeight:2 }}>
            <Row label="지표명"  val={lawModal.label} />
            <Row label="근거법령" val={`중대재해처벌법 ${lawModal.law}`} />
            <Row label="이행주기" val={lawModal.period} />
            <Row label="목표횟수" val={`${lawModal.target}회`} />
            <div style={{ marginTop:12,padding:12,background:C.surface2,borderRadius:8,fontSize:12,color:C.textSub,lineHeight:1.8 }}>
              중대재해처벌법 제4조에 따라 경영책임자는 해당 의무를 성실히 이행하여야 하며, 미이행 시 동법 제6조에 따른 형사처벌 대상이 될 수 있습니다.
            </div>
          </div>
        </Modal>
      )}

      {/* ── 경영책임자 종합 보고서 ── */}
      {reportModal && (
        <MasterReportModal indData={indData} setIndData={setIndData} company={company} onClose={()=>setReportModal(false)} />
      )}
    </div>
  );
}

//  이행기록 추가 모달
//  내용(필수) + 파일첨부(선택)
// ═══════════════════════════════════════════════
function RecordAddModal({ ind, currentCount, onAdd, onClose }) {
  const [content, setContent] = useState("");
  const [file,    setFile]    = useState(null);
  const [confirm, setConfirm] = useState(false); // 최종 확인 단계

  const handleSelect = (e) => { if(e.target.files[0]) setFile(e.target.files[0]); };

  const handleSubmit = () => {
    if (!content.trim()) { alert("이행 내용을 입력하세요."); return; }
    setConfirm(true);
  };

  if (confirm) return (
    <Modal onClose={()=>setConfirm(false)} title="✅ 이행 기록 최종 확인">
      <div style={{ fontSize:13,lineHeight:2 }}>
        <Row label="지표"   val={ind.label} />
        <Row label="이행내용" val={content} />
        <Row label="첨부파일" val={file?.name||"없음"} />
        <Row label="등록일"  val={today()} />
        <Row label="이행 횟수" val={`${currentCount+1}/${ind.target}`} />
        <div style={{ padding:10,background:C.warningDim,borderRadius:8,border:`1px solid ${C.warning}44`,fontSize:12,color:C.warning,marginTop:8 }}>
          ⚠️ 등록 후에는 마스터만 삭제 가능합니다. 내용을 다시 한번 확인하세요.
        </div>
        <div style={{ display:"flex",gap:8,marginTop:14 }}>
          <button onClick={()=>onAdd(content.trim(),file)} style={{ ...BS,background:C.success,color:"#fff",flex:1,fontWeight:800 }}>✓ 등록 확정</button>
          <button onClick={()=>setConfirm(false)} style={{ ...BS,background:C.surface2,flex:1 }}>수정</button>
        </div>
      </div>
    </Modal>
  );

  return (
    <Modal onClose={onClose} title={`✏️ 이행 기록 추가 — ${ind.label}`}>
      <div style={{ fontSize:12,color:C.textMuted,marginBottom:10 }}>
        목표 {ind.target}회 중 현재 {currentCount}회 완료 · <b style={{ color:C.warning }}>{ind.target-currentCount}회 남음</b>
      </div>
      <div style={{ marginBottom:10 }}>
        <div style={{ fontSize:12,fontWeight:700,marginBottom:6,color:C.text }}>이행 내용 <span style={{ color:C.danger }}>*</span> <span style={{ color:C.textMuted,fontWeight:400 }}>(필수 입력)</span></div>
        <textarea value={content} onChange={e=>setContent(e.target.value)}
          placeholder={`예) ${ind.period} ${ind.label} 실시 완료. 참석자: 홍길동 외 5명...`}
          style={{ ...IS,height:90,resize:"vertical" }} />
      </div>
      <div style={{ marginBottom:10 }}>
        <div style={{ fontSize:12,fontWeight:700,marginBottom:6,color:C.text }}>증빙 파일 첨부 <span style={{ color:C.textMuted,fontWeight:400 }}>(선택)</span></div>
        <label style={{ display:"block",padding:"14px",border:`2px dashed ${file?C.success:C.border}`,borderRadius:8,textAlign:"center",cursor:"pointer",background:file?C.successDim:C.surface2 }}>
          {file ? <><div style={{ color:C.success,fontWeight:700 }}>📎 {file.name}</div><div style={{ fontSize:11,color:C.textMuted,marginTop:4 }}>클릭하여 변경</div></>
                : <><div style={{ color:C.textMuted }}>📁 클릭하여 파일 선택</div><div style={{ fontSize:11,color:C.textMuted,marginTop:4 }}>PDF, 이미지, 문서 등</div></>}
          <input type="file" style={{ display:"none" }} onChange={handleSelect} />
        </label>
      </div>
      <div style={{ display:"flex",gap:8,marginTop:4 }}>
        <button onClick={handleSubmit} style={{ ...BS,background:C.accent,color:"#000",fontWeight:800,flex:1 }}>다음 → 최종 확인</button>
        <button onClick={onClose} style={{ ...BS,background:C.surface2,flex:1 }}>취소</button>
      </div>
    </Modal>
  );
}

function AIReportModal({ report, onClose, onSave }) {
  const [result,setResult]=useState(""); const [loading,setLoading]=useState(false);
  useEffect(()=>{
    (async()=>{
      setLoading(true);
      try {
        const res=await fetch("https://api.anthropic.com/v1/messages",{method:"POST",headers:{"Content-Type":"application/json"},body:JSON.stringify({model:"claude-sonnet-4-20250514",max_tokens:1000,messages:[{role:"user",content:`건설/전력공사 현장 안전관리자입니다. 아래 위험요인에 대해 즉시 실천 가능한 핵심 예방 대책 3가지를 개조식(bullet)으로만 답하세요. 법조항이나 서론 없이 실무 조치만:\n\n위험요인: ${report.factor}\n위치: ${report.location}\n위험도: ${report.severity}`}]})});
        const d=await res.json(); setResult(d.content?.[0]?.text||"오류");
      } catch { setResult("AI 연결 오류"); }
      setLoading(false);
    })();
  },[]);
  return (
    <Modal onClose={onClose} title="🤖 AI 안전 조치 제안">
      <div style={{ fontSize:12,marginBottom:10,color:C.textSub }}><b>위험요인:</b> {report.factor}</div>
      {loading ? <div style={{ textAlign:"center",padding:24,color:C.accent }}>⚙️ AI 분석 중...</div>
        : <div style={{ background:C.surface2,borderRadius:8,padding:14,fontSize:13,lineHeight:1.9,whiteSpace:"pre-wrap" }}>{result}</div>}
      {!loading&&result&&(
        <div style={{ display:"flex",gap:8,marginTop:12 }}>
          <button onClick={()=>onSave(report.id,result)} style={{ ...BS,background:C.accent,color:"#000",flex:1 }}>저장</button>
          <button onClick={onClose} style={{ ...BS,background:C.surface2,flex:1 }}>닫기</button>
        </div>
      )}
    </Modal>
  );
}

function AIPage({ company }) {
  const [policyRes,setPolicyRes]=useState(""); const [pLoading,setPLoading]=useState(false);
  const [chatInput,setChatInput]=useState(""); const [chatHist,setChatHist]=useState([]); const [cLoading,setCLoading]=useState(false);
  const chatEndRef=useRef(null);
  useEffect(()=>chatEndRef.current?.scrollIntoView({behavior:"smooth"}),[chatHist]);

  const genPolicy=async()=>{
    setPLoading(true); setPolicyRes("");
    try { const r=await fetch("https://api.anthropic.com/v1/messages",{method:"POST",headers:{"Content-Type":"application/json"},body:JSON.stringify({model:"claude-sonnet-4-20250514",max_tokens:1000,messages:[{role:"user",content:`${company} 전력·건설 회사를 위한 중대재해처벌법 제4조 요건에 부합하는 안전보건 경영방침 초안을 전문 공식 문체로 작성해 주세요. 400자 이내.`}]})}); const d=await r.json(); setPolicyRes(d.content?.[0]?.text||"오류"); } catch { setPolicyRes("AI 연결 오류"); }
    setPLoading(false);
  };

  const sendChat=async()=>{
    if(!chatInput.trim()) return;
    const msg=chatInput; setChatInput("");
    const nh=[...chatHist,{role:"user",content:msg}]; setChatHist(nh); setCLoading(true);
    try { const r=await fetch("https://api.anthropic.com/v1/messages",{method:"POST",headers:{"Content-Type":"application/json"},body:JSON.stringify({model:"claude-sonnet-4-20250514",max_tokens:1000,system:`당신은 중대재해처벌법 전문 안전보건 AI 비서입니다. ${company} 소속 실무자를 위해 간결하고 실용적인 답변을 제공합니다.`,messages:nh})}); const d=await r.json(); setChatHist([...nh,{role:"assistant",content:d.content?.[0]?.text||"오류"}]); } catch { setChatHist([...nh,{role:"assistant",content:"AI 연결 오류"}]); }
    setCLoading(false);
  };

  return (
    <div>
      <SecHead title="🤖 AI 안전비서" sub="Claude AI 기반 안전보건 지원 솔루션" />
      <div style={{ background:C.surface,border:`1px solid ${C.border}`,borderRadius:12,padding:16,marginBottom:20 }}>
        <div style={{ fontSize:13,fontWeight:700,marginBottom:8 }}>✨ 경영방침 초안 자동 생성</div>
        <button onClick={genPolicy} disabled={pLoading} style={{ ...BS,background:C.accent,color:"#000",fontWeight:800,width:"100%" }}>{pLoading?"⚙️ 생성 중...":"🚀 경영방침 초안 생성"}</button>
        {policyRes&&<div style={{ marginTop:12,padding:14,background:C.surface2,borderRadius:8,fontSize:13,lineHeight:1.9,whiteSpace:"pre-wrap" }}>{policyRes}</div>}
      </div>
      <div style={{ background:C.surface,border:`1px solid ${C.border}`,borderRadius:12,padding:16 }}>
        <div style={{ fontSize:13,fontWeight:700,marginBottom:8 }}>💬 안전보건 AI 상담</div>
        <div style={{ minHeight:200,maxHeight:320,overflowY:"auto",padding:8,background:C.surface2,borderRadius:8,marginBottom:10,display:"flex",flexDirection:"column",gap:8 }}>
          {chatHist.length===0&&<div style={{ color:C.textMuted,fontSize:12,textAlign:"center",paddingTop:60 }}>중대재해처벌법, 위험성 평가, 안전보건 실무 등을 질문하세요.</div>}
          {chatHist.map((m,i)=>(
            <div key={i} style={{ alignSelf:m.role==="user"?"flex-end":"flex-start",maxWidth:"85%",background:m.role==="user"?C.accentDim:C.surface3,padding:"8px 12px",borderRadius:10,fontSize:12,lineHeight:1.7,whiteSpace:"pre-wrap" }}>{m.content}</div>
          ))}
          {cLoading&&<div style={{ color:C.accent,fontSize:12 }}>⚙️ 답변 생성 중...</div>}
          <div ref={chatEndRef} />
        </div>
        <div style={{ display:"flex",gap:8 }}>

        </div>
      </div>
    </div>
  );
}

  if(diff<=90) return <span style={{...BDG,background:"#d97706",color:"#fff"}}>D-{diff}</span>;
  return <span style={{...BDG,background:"#1e3a5f",color:"#60a5fa"}}>D-{diff}</span>;
